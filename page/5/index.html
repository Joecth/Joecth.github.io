\begin{array}{l}
{\bf for~} i := 0, \ldots , M-1 \

~ {\bf for} j := 0, \ldots , N-1 \

~ ~ {\rm Load~} C_{i,j} \mbox{ into registers} \

~ ~ {\bf for~} p := 0, \ldots , K-1 \

~ ~ ~ {\rm Load} A_{i,p} , {\rm and} B_{p,j}
\mbox{ into registers} \

~ ~ ~ C_{i,j} := A_{i,p} B_{p,j} + C_{i,j} \

~ ~ {\bf end} \

~ ~ {\rm Store~} C_{i,j} \mbox{ to memory} \

{\bf end}
\end{array}
\end{equation*}

  • 定義、出口、拆解

    • def dfs(root):

      if not root:
      return [0, 0]
  left_M_D, left_M_C = self.dfs(root.L)
  right_M_D, right_M_C = self.dfs(root.R)
  M_D = max(left_M_D, right_M_D, left_M_C + right_M_C)
  M_C = max(right_M_C, right_M_C) + 1

      ​ return [M_D, M_C]

```python
class Solution:
   def romanToInt(self, s: str) -> int:
   # check input
    if not s:
        return -1
		# build a symbol:value dict
    d = {'I':1, 'V':5, 'X':10, 'L':50, 'C':100, 'D':500, 'M':1000}
    n = len(s)
    
    result = 0
    
    # iterate characters in s
    for idx in range(n-1):
        char_curr = s[idx] # e.g. I in IX
        char_next = s[idx+1] # OOR..
        if d[char_curr] < d[char_next]:
            result = result - d[char_curr]
        else:
            result = result + d[char_curr]
    
    result = result + d[s[-1]]
    return result
```

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        if not l1:
            return -1
        if not l2: 
            return -1
        
        result = ListNode(0)
        curr = result
        total = 0
        while l1 or l2 or total:
            if l1:
                total += l1.val
                l1 = l1.next
            if l2:
                total += l2.val
                l2 = l2.next
            out = (total) % 10
            curr.next = ListNode(out)
            curr = curr.next
            
            carry = total // 10
            total = carry
        
        return result.next

    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        # check input
        if not l1:
            return l2
        if not l2: 
            return l1
        
        dummy = ListNode(0)
        cur = dummy
        carry = 0
        
        while l1 or l2 :
            if l1:
                carry += l1.val
                l1 = l1.next
            if l2:
                carry += l2.val
                l2 = l2.next
                
            cur.next = ListNode(carry%10)
            cur = cur.next
            carry //= 10
        return dummy.next
```
